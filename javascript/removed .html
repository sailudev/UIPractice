    • `; // Add the new list item HTML to the task list taskList.innerHTML += listItemHTML; // Increment the serial number serialNumber++; // Clear the input field taskInput.value = ""; } function removeTask(serial) { // Remove the corresponding list item based on the serial number var taskListItem = document.getElementById(`task-${serial}`); taskListItem.remove(); }
